	\begin{enumerate}
		
		\item
			已知关于$x$的不等式$kx^2-kx-2>0$的解集是空集，求实数$k$的取值范围.

			~\\
			\begin{enumerate} [$1^\circ$]
				\item
					$k=0$时，

					原不等式可化简为$-2>0$.

					此时有$x\in \emptyset$，符合题意.

				\item
					$k\neq 0$时，

					解不等式组

					$$
						\left\{
						\begin{array} {c}
							k<0\\
							\Delta=k^2+8k<0\\
						\end{array}
						\right.\Rightarrow-8\leq k<0.
					$$

				综上，$k\in [-8,0].$

			\end{enumerate}

	\end{enumerate}
